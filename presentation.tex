\section{Ausgangssituation}
\frame {
    \frametitle{Ausgangssituation}
    \begin{itemize}
        \item Kryptographische Sicherheit auf Transportebene $\Rightarrow$ \gls*{https}
        \item Verschiedene Autoren verfassen Inhalte auf einer Website
        \item Autoren verfassen Inhalte auf verschiedenen Websites
        \item Inhalte können nicht mit kryptographischen Merkmalen versehen werden
    \end{itemize}
}

\section{Forschungsfragen}
\frame {
    \frametitle{Forschungsfragen}
    \begin{itemize}
        \item Wie können digitale Signaturen in \gls*{xhtml} integriert werden, ohne dass dieser W3C-Standard dabei verletzt wird?
        \item Welche Möglichkeiten bietet \gls*{xml-dsig} zur wahlfreien Selektion von Bereichen innerhalb eines XHTML-Dokuments?
        \item Wie können die öffentlichen Schlüssel zu den Signaturen verteilt werden? Welche der bestehenden Technologien eignen sich dafür am besten?
    \end{itemize}
}

\section{Aufbau}
\frame {
    \frametitle{Aufbau}
    \begin{itemize}
        \item Einführung in \gls*{xml-dsig}
        \item Verbindungen zwischen \gls*{xml-dsig} und \gls*{xhtml}
        \item Signaturverfahren für \gls*{xml-dsig}
        \item Anforderungen an ein Browser-AddOn (\gls*{firefox})
    \end{itemize}
}

\section{Erkenntnisse}
\frame {
    \frametitle{Erkenntnisse}
    \begin{itemize}
        \item \gls*{xml-dsig} bietet große Flexibilität beim Signieren von \gls*{xml}-Dokumenten mit \gls*{x509} oder \gls*{openpgp}
        \item Mehrere Ansätze für Signatur-Bindung\\
            Hervorzuheben sind:
            \begin{itemize}
                \item Mikro-Formate mit fixen Signaturvorgaben
                \item \xmlelem{link}-Tag für vollständige Ausnutzung von \gls*{xml-dsig}
            \end{itemize}
        \item Signaturverfahren:
            \begin{itemize}
                \item \gls*{x509}: für Unternehmerischen oder staatlichen Einsatz geeignet
                \item \gls*{openpgp}: für schnelle Adaption durch Privatbenutzer geeignet, höherer Grad an Automation möglich
            \end{itemize}
    \end{itemize}
}

\section{Conclusio}
\frame {
    \frametitle{Conclusio}
    \begin{itemize}
        \item Weder \gls*{openpgp} noch \gls*{x509} decken alle Erfordernisse vollständig ab
        \item Einsatz einer Abstraktionsschicht um beide Verfahren zu ermöglichen
        \item Definition des Verfahrens in der Signatur selbst
    \end{itemize}
}

\section{Ausblick}
\frame {
    \frametitle{Ausblick}
    \begin{itemize}
        \item Entwicklung eines Prototypen für ein Browser-AddOn in \gls*{firefox}
        \item Zumindest Unterstützung für \gls*{openpgp} oder \gls*{x509}
    \end{itemize}
}
